#include <stdio.h>

int main() {
    
    char blood;
    
    printf("혈액형 입력(A/B/O): ");
    scanf("%c", &blood);
    
    switch(blood) {
        case 'A': printf("A\n"); break;
        case 'B': printf("B\n"); break;
        case 'O': printf("O\n"); break;
        
        default: printf("잘못된 입력\n"); break;
    }
    
    return 0;
}
