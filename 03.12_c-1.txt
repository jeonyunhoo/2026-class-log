#include <stdio.h>

int main() {
    
    int num;
    
    while(1) { //반복, 0 -> 거짓, 1 -> 참
               //무한반복
    
        printf("숫자를 입력하세요!");
        scanf("%d", &num);
    
        if(num == 10) {
            printf("정답입니다!\n");
            break;
        } else if(num >= 10) {
            printf("너무 높아요!\n");
        } else {
            printf("너무 낮아요!\n");
        }
    
    }
    return 0;
}
