#include <stdio.h>

int main() {
    
    int num;
    
    printf("정수 입력: ");
    scanf("%d", &num);
    
    if(num % 2 == 1) { //if란 조건문 ()필수 %는 나머지를 의미 {}if의 범위 규정
        printf("홀수");
        printf("\n2로 나눈 나머지가 1");
    } else { //if에 포함되지 않는 값들을 처리한다.
        printf("짝수");
        printf("\n2로 나눈 나머지가 0");
    }
    
    return 0;
}

// >, <, <=, >= 대소 비교, == 같다, = 좌변에 우변을 대입한다, != 같지 않다
